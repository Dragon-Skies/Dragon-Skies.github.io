C++普及数据结构模板

1. 线性表

1.1.栈

struct stack{                        
	int tot;
   node s[MAXN];
	stack(){tot=0;}
	bool empty(){return tot==0;}
	int size(){return tot;}
	node top(){return s[tot];}
	int pop(){
		if(empty()) return 0;
		else{return tot--;} 
	}
	void push(int x){
		s[++tot]=x;
	}
};

1.2.队列

struct queue{
    int l,r,cnt;node qu[N];
    node front(){return qu[l];}
    void ne(int &x){++x;if(x==N) x=1;}
    void push(node x){ne(r);qu[r]=x;++cnt;}
    void pop(){ne(l);--cnt;}
    bool empty(){return !cnt;}
    int size(){return cnt;}
    void clear(){r=cnt=0,l=1;}
};

1.3.双向队列

struct deque{
    int l,r,cnt;node qu[N];
    deque(){r=cnt=0,l=1;}
    node front(){return qu[l];}
    node back(){return qu[r];}
    void ne(int &x){++x;if(x==N) x=1;}
    void la(int &x){--x;if(!x) x=N-1;}
    void push_front(node x){la(l);qu[l]=x;++cnt;}
    void push_back(node x){ne(r);qu[r]=x;++cnt;}
    void pop_front(){ne(l);--cnt;}
    void pop_back(){la(r);--cnt;}
    bool empty(){return !cnt;}
    void size(){return cnt;}
    void clear(){r=cnt=0,l=1;}
};

1.4.链表

struct List{
	int s,ne,la;
	List(int q=0,int n=-1,int p=-1){s=q;ne=n;la=p;}
};
vector <List> l(1);
void push(int x){
	int tot=l.size();
	l[tot-1].ne=tot;
	l.push_back(List(x,-1,tot-1));
}
void pop(int x){
	l[x-1].ne = x+1;
	l[x+1].la = x-1;
}

1.5.并查集

struct unionFind{
	int bin[MAXN];
	unionFind(){for(int i=1;i<=5000;++i) bin[i]=i;}
	int find(int x){
		if(bin[x]==x) return x;
		return find(bin[x]);
	}
	bool ask(int x,int y){
		if(find(x)==find(y)) return true;
		return false;
	}
	int insert(int x,int y){return bin[find(y)]=find(x);}
	int del(int x){bin[x]=0;}
};

2.树

2.1. （大根）堆（优先队列）

#define swap(a,b) int t=a;a=b;b=t;
struct heap{
	int w[N];
	int tot;
	int top(){return w[1];}
	void modify(int x){
		if(x==1) return ;
		if(w[x]>w[x>>1]){
			swap(w[x],w[x>>1]);
			modify(x>>1);
		}
	}
	void push(int k){
		w[++tot]=k;
		modify(tot);
	}
	void repair(int x){
		int tar=w[x<<1]>w[x<<1|1]?x<<1:x<<1|1;
		if(w[x]<w[tar]){
			swap(w[x],w[tar]);
			repair(tar);
		}
	}
	void pop(){
		swap(w[1],w[tot]);
		w[tot--]=0;
		repair(1);
	}
}h;

2.2. 二叉树

struct node{
	int l,r,dad;
	node(){l=r=dad=-1;}
};
vector <node> t(1);
void add(int u,int a,int b){
	if(t.size()-1<max(max(u,a),b)) t.resize(max(max(u,a),b)+1);
	if(a!=0) t[u].l=a,t[a].dad=u;
	if(b!=0) t[u].r=b,t[b].dad=u;
	return;
}

3.图论

1.邻接表存图

struct edge{
	int to,w;
	edge(int _to=-1,int _w=-1){to=_to;w=_w;}
};
vector <edge> G[N];
inline void init(int n){
	for(int i=1;i<=n;i++) G[i].resize(1);
	return;
}
inline void add(int u,int v,int w){
	G[u].push_back(edge(v,w));
	G[v].push_back(edge(u,w));
	return ;
}
