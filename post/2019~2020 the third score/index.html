import xlrd
import xlwt
import os


def open_excel(file='test.xlsx'):
    try:
        data = xlrd.open_workbook(file)
        return data
    except:
        pass


def read_excel(filename):
    wd = open_excel(filename)
    table = wd.sheets()[0]
    row = table.nrows
    col = table.ncols
    List = []
    for i in range(row):
        List.append([])
        for j in range(col):
            List[i].append(table.cell(i, j).value)
    return List


def write_excel(namelist, savename, savebool=True):
    # namelist格式要求,为二维数组,namelist[i]表示第i+1行需要写入的内容
    row = len(namelist)
    col = len(namelist[0])
    workbook = xlwt.Workbook()
    worksheet = workbook.add_sheet('New Sheet')
    for i in range(row):
        for j in range(col):
            worksheet.write(i, j, namelist[i][j])
    if(savebool):
        workbook.save(savename)
    else:
        return workbook

def count(al,now):
    if(now>=al*0.9):
        return (now-(al*0.8)+1)*20
    if(now>=al*0.8):
        return (now-(al*0.8)+1)*10
    if(now<=al*0.7):
        return (now-al*0.7)*100
    return 0

def re_list_to_dict(List):
    now = {}
    for i in List:
        tmp = i[:]
        tmp.pop(0)
        now[i[0]] = tmp
    return now
