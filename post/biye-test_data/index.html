for i in range(10,21):

from cyaron import *

	ans=1

	return ans

for i in range(19,20):

		test.input_write(random_int(1,100000))
	test.input_write('\n')
	for j in range(m):
		a = random_int(1,5)
		if(a==1 or a==2):
			x = random_int(1,n-1)
			y = random_int(1,n-x+1)
			test.input_write(str(random_int(1,100000))+' 3 '+str(x)+' '+str(y)+'\n')
		elif(a==3 or a==4):
			x = random_int(1,n-1)
			y = random_int(1,n-x+1)
			test.input_write(str(random_int(1,100000))+' 1 '+str(x)+' '+str(y)+' '+str(random_int(1,1000))+'\n')
		else:
			x = random_int(1,n-1)
			y = random_int(1,n-x+1)
			test.input_write(str(random_int(1,100000))+' 2 '+str(x)+' '+str(y)+' '+str(random_int(1,1000))+'\n')
	t=1

		if(t>=100000):
			t = 100000

#include<bits/stdc++.h>
using namespace std;
struct unionFind{
	int bin[100005];
	unionFind(int n){
		for(int i=1;i<=n;++i) bin[i]=i;
	}
	int find(int x){
		if(bin[x]==x) return x;
		return find(bin[x]);
	}
	int insert(int x,int y){
		return bin[find(y)]=find(x);
	}
};
int n,x,y,m;
int main(){
	ios::sync_with_stdio(false);
	cin>>n>>m;
	unionFind u(n);
	for(int i=1;x!=-1&&y!=-1;++i){
		cin>>x>>y;
		if(x==-1||y==-1) break;
		u.insert(x,y);
	}
	for(int i=1;i<=n;++i)
		if(u.find(i)==i) cout<<i<<endl;
	for(int i=1;i<=m;++i){
		cin>>x;
		if(u.find(x)!=x) cout<<u.find(x)<<endl;
		else cout<<"0"<<endl;
	}
	return 0;
}

#include<bits/stdc++.h>
using namespace std;
struct node{
	int id,w,p;
}x,y;
int n,cnt;
queue <node> q;
node a[1030];
int main(){
	cin>>n;
	for(int i=1;i<=1<<n;i++){
		cin>>a[i].w;a[i].id=i;q.push(a[i]);
	}
	while(q.size()>1){
		for(int i=1;i<=1<<(n-cnt-1);i++){
			x=q.front();q.pop();y=q.front();q.pop();
			if(x.w>y.w){a[y.id].p=cnt+1,q.push(x);}
			else{a[x.id].p=cnt+1,q.push(y);}
		}
		cnt++;
	}
	a[q.front().id].p=n+1;
	for(int i=1;i<=1<<n;i++) printf("%d ",a[i].p);
	return 0;
}

【T4】 Sakura

#include<bits/stdc++.h>
const int ha=5201314;
using namespace std;
struct node{
	int tim,pox,x,y,w;
};
struct sakura{
	long long s;
	int last,next;
	inline void init(int i,int n){
		if(i==1) last=-1;
		else last=i-1;
		if(i==n) next=-1;
		else next=i+1;
	}
};
sakura sak[100005];
inline void add(int x,int y){
	sak[sak[x].last].next = sak[x].next;
	sak[sak[x].next].last = sak[x].last;
	sak[x].last = y;
	sak[x].next= sak[y].next;
	sak[sak[y].next].last = x;
	sak[y].next = x;
	return;
}
node q[100005];bool vis[100005];
int n,m,k,head=1;
inline void push_down(int x=head){
	if(q[x].pox==1){
		int u=q[x].x; 
		for(int i=1;i<=q[x].y;i++){
			sak[u].s=(sak[u].s+q[x].w)%ha;
			u=sak[u].next;
			if(u==-1) return;
		}
		return;
	}
	if(q[x].pox==2){
		int u=q[x].x; 
		for(int i=1;i<=q[x].y;i++){
			sak[u].s=(sak[u].s*q[x].w)%ha;
			u=sak[u].next;
			if(u==-1) return;
		}
		return;
	}
	if(q[x].pox==3){
		add(q[x].x,q[x].y);
		return;
	}
}
inline bool cmp(node a,node b){
	if(a.tim!=b.tim) return a.tim<b.tim;
	return a.pox<b.pox;
}
int main(){
	ios::sync_with_stdio(false);
	cin>>n>>m>>k;
	for(int i=1;i<=n;i++){
		cin>>sak[i].s;
		sak[i].init(i,n);
	}
	for(int i=1;i<=m;i++){
		cin>>q[i].tim>>q[i].pox>>q[i].x>>q[i].y;
		if(q[i].pox!=3) cin>>q[i].w;
	}
	sort(q+1,q+m+1,cmp);
	for(int i=1;i<=k;i++){
		int ti,a,b,c,d;
		long long cf=0,yue=0,ming=0;
		cin>>ti>>a>>b>>c>>d;
		while(head<=m&&q[head].tim<=ti)
			push_down(head++);
		int u=a;
		for(int j=1;j<=b+1;j++){
			yue=(yue+sak[u].s)%ha;
			vis[u]=true;
			u=sak[u].next;
			if(u==-1) break;
		}
		u=c;
		for(int j=1;j<=d+1;j++){
			if(vis[u]) cf=(cf+sak[u].s)%ha;
			else ming=(ming+sak[u].s)%ha;
			u=sak[u].next;
			if(u==-1) break;
		}
		cout<<((cf*(yue-1)%ha)+(ming*yue%ha))%ha<<endl;
		memset(vis,0,sizeof(vis)); 
	}
	return 0;
}
